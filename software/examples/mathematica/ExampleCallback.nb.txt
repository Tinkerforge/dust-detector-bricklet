Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
dd=NETNew["Tinkerforge.BrickletDustDetector",uid,ipcon]
ipcon@Connect[host,port]

(* Set Period for dust density callback to 1s (1000ms) *)
(* Note: The dust density callback is only called every second if the *)
(*       dust density has changed since the last call! *)
dd@SetDustDensityCallbackPeriod[1000]

(* Callback function for dust density callback (parameter has unit µg/cm^3) *)
DustDensityCB[sender_,dust_density_]:=
 Print["Dust Density: "<>ToString[N[Quantity[dust_density,"µg/cm^3"]]]]
AddEventHandler[dd@DustDensity,DustDensityCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[dd]
ReleaseNETObject[ipcon]
